[Xosé_María_Costa][Poesía contemporánea]
[inicio]
e_ra un tem_po de de_sas_tres e de or_xí_as,
de be_a_tas te_me_ra_tas,
de i_lu_mi_ña_dos be_o_dos
e de ac_has se_mo_ven_tes,
e_ra un tem_po de tou_rei_ros e de o_lés,
de te_mo_res e de aies,
e_ra o no_so tem_po
de par_ti_das e de a_za_res,
de fut_bo_lis_tas e de im_pe_rios,
vi_ví_a_se nun si_len_cio de sé_cu_los,
nun mar_mu_rio ca_la_do
con fú_ne_bres es_tam_pas ber_mel_las
nas so_le_o_sas bei_ras da es_tra_da,
e_ra o tem_po da có_pu_la en_tre o ver_bo e a pa_la_bra,
im_pe_rial or_gas_mo ho_mo_se_xual,
ti ti_rí_as un xes_to se_cu_lar de «ni_ña pi_ja»,
e_ra un tem_po de pro_me_sas e de xu_gos
que nos sal_fe_rí_u a to_dos
e den_de aq_ue_la le_va_mos a mar_ca,
a no_sa que_ri_da e co_é_ta_ne_a com_pa_ñei_ra,
nun an_dar zig_zag_ue_an_te,
li_bi_di_no_so, ul_tra e ca_ver_na_rio
que ol_la_mos de lon_xe e de cer_ca
co_ma un lar_ga_cí_o luar que nos dis_te_ra.
[fin]
