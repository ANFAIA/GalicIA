[i_ni_cio]

a quen_tar_se a_o sol ma_re_lo:

a lon_gas ter_ras do sul,

an_tre la_van_dei_ras do_nas,
pa_pos-ru_bios es_cu_dei_ros,
e cu_ru_xas a_xe_xan_tes,
e ga_los de cra_rín re_xo.
. . . . . . . . . . . . . . . . . . . . . . . .

che_gó_u o ven_to do in_ver_no,
co_a ne_ve por bran_ca bes_ta,

côs guer_rei_ros das fo_ler_pas

por na_mo_rar â ou_to_ní_a,
o seu pa_zo de sol_tei_ro
lei_xa_ra, en ter_ras lon_xa_nas
de a_gros de bran_cor e_ter_no.
moi_ta mor_tan_dá de ver_des

na ca_ti_va grei das her_bas,
na ou_tra grei dos ar_bo_re_dos,
des_fol_ló_u ro_sas don_ce_las
e mur_chó_u froi_tos ber_mel_los.
. . . . . . . . . . . . . . . . . . . . . . . .
dí_xol_le â lou_ra ou_to_ní_a:

—meu pa_zo de ca_ba_lei_ro,
tris_te nos a_gros do nor_te,
a_gar_da o teu cor_po be_lo;
que_ro e_se teu mor_no cor_po

que_ro o teu té_pe_do a_len_to;
pa_ra go_ren_tar meus ol_los
que_ro os teus lou_ros ca_be_los;
que_ro que quen_tes a bi_cos
a fria_xe do meu pei_to.

dí_xol_le a lou_ra ou_to_ní_a:

—lei_xá_de_me, ca_ba_lei_ro;
se de no_bre vos ga_ba_des
res_pei_ta_de o meu so_se_go;
son ca_sa_da, de ma_ri_do,
co_ma vos, no_bre e guer_rei_ro
que loi_ta en ter_ras lon_xa_nas

fi_del se_rei a_ta a mor_te
a_o meu ho_me, pois xu_rei_no;
se de no_bre vos ga_ba_des
lei_xá_de_me, ca_ba_lei_ro.

tei_mo_so, â lou_ra ou_to_ní_a,
dí_xol_le o ven_to do in_ver_no:

—te_ño se_de de a_lou_mi_ños
se non mos das, rou_ba_reic_hos;
que_re a re_ga da tua ri_sa
a mur_char fror do meu pei_to.
lei_xa a_o teu ho_me, fer_mo_sa,
que el es_tá lon_xe i-eu per_to;
vei que, sen_llei_ra, lei_xou_te
o teu a_mor es_que_cen_do
e no ha vol_ver, que si vol_ta;
por teu a_mor, ma_ta_rei_no
ar_man_do o meu for_te bra_zo
co_a mi_ña es_pa_da de xe_o;
lei_xa a_o teu ho_me, fer_mo_sa,
non des_pre_ces os meus pre_gos.

dí_xol_le a lou_ra ou_to_ní_a:

—vos non so_des ca_ba_lei_ro
se non trê_i_dor des_lei_ga_do.
fi_del a_o meu xu_ra_men_to
de_nan_tes mor_te me de_ra
que ou_vir tan no_xen_tos pre_gos.
. . . . . . . . . . . . . . . . . . . . . . . .
in_da que a_si_sa_das ver_bas
di_xe_ra a ou_to_ní_a a_o ven_to,
o ven_to, to_lo de a_mo_res
e po_la car_ra_xe ce_go,

a_bo_fé, non foi do_a_da
a ga_ñan_za do cas_te_lo;

a_bo_fé non foi do_a_da,

e cru_del ven_to do nor_te,
do trê_i_dor ven_to do in_ver_no.
por non vi_vir pri_sio_nei_ra
do trun_fa_dor ca_ba_lei_ro

do cu_me do seu cas_te_lo;

fu_xi_ron cho_ran_do a re_gos;
as coi_ta_das la_van_dei_ras
a_os trun_fa_do_res se de_ron.
. . . . . . . . . . . . . . . . . . . . . . . .

. . . . . . . . . . . . . . . . . . . . . . . .

o coi_ta_do ca_ba_lei_ro.
