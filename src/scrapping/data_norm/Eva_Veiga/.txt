tan so_a_men_te sós de nós mes_mos  
que an_da_mos a bus_cár_mo_nos nos ou_tros  
e e_les nun_ca nos vi_ron pa_sar.

é a can_sei_ra in_fla_ma_da  
dun tem_po sen ás.

ar_ras_tran os seus pa_sos nun_ha dan_za  
som_brí_a de ro_sas.

tran_se_ún_tes to_dos na pe_ri_fe_ria dun re_lo_xio  
de al_ta pre_ci_si_ón.  
meu a_mor,  
es_tou a mor_rer e non me es_pe_ras.  

des_ter_ro de bol_bo_re_tas:  
des_fei_to o vo_o  
co_ma som_bra de ai_re.

co seu po de dia_man_te.  
ce_gá_ra_no-los ol_los  
a sú_a lim_pa tur_bu_len_cia.  

que sig_nos mur_mu_ra_ron o cre_pús_cu_lo  
dos ven_tos

a_fou_tos de dor,  
ou_sa_ron pe_ne_trar no sa_cro es_pa_cio  
da san_ta com_pa_re_cen_cia.  

é po_si_ble que ou_tra for_ma hu_ma_na  
nos a_gar_de no fu_tu_ro.  
é po_si_ble que a_gar_de  
a de_fi_ni_ti_va te_gra que a es_cri_ba.  
men_tres, a_van_za_mos len_ta_men_te,  
re_sol_ve_mos ou en_re_da_mos.  
so_ña_mos.  
to_do se_mel_la em_pur_rar_nos ca_ra a_lí,  
ou den_de a_lí al_go pa_re_ce a_tra_er_nos  
cun_ha lei de gra_vi_ta_ci_ón des_co_ñe_ci_da.  

ven_ci_dos os im_nu_me_ra_bles bra_zos de deus,  
so_bre a al_ma fi_ca un_ha pe_sa_du_me an_ti_ga:

di_cir pa_ra ser.

nos se_men_ta no ter_ri_to_rio es_cu_ro

por on_de san_gra  
o es_pa_cio au_men_ta, flú_e xa sen tem_po,
