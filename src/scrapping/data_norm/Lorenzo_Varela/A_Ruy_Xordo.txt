[i_ni_cio]
a_ra_mos so_bre os mor_tos nes_ta ter_ra
i o no_so pan ten un sa_bor de ho_sos
fa_mi_lia_res, ir_máns. o mon_te ber_ra
bai_xo do a_ra_do, e chá_man_nos os no_sos

des_de a mor_te con vo_ces co_no_ci_das:
«nin mar_que_ses, nin cre_gos, nin dou_to_res
fi_xe_ron as ri_bei_ras ver_de_ci_das,
nin o guer_rei_ro coi_ro dos tam_bo_res.

os con_da_dos do pol_vo son dos mor_tos
e quen quei_ra ser do_no des_ta ter_ra
que ve_ña na_ve_gan_do a_os no_sos por_tos.

os que pi_dan o fru_to sin la_bo_res,
si non mor_ren de seu, mor_ran de guer_ra,
e des_ta ter_ra, a_sí, se_rán se_ño_res».
[fin]
