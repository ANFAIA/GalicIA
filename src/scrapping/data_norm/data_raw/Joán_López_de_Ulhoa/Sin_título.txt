[Joán_López_de_Ulhoa][Trovadorismo]
[inicio]
eu nun_ca dor_mio na_da,
cui_dand'en meu a_mi_go;
el que tan mui_to tar_da,
se outr'a_mor ha_si_go,
er_go o meu, que_rí_a
mor_rer hoj'es_te dí_a.

e cuid'en es_to sem_pre:
non sei que de mí se_ja;
el que tan mui_to tar_da,
se ou_tro ben de_se_ja,
er_go o meu, que_rí_a
mor_rer hoj'es_te dí_a.

se o faz, faz-mi tor_to
e, par deus, mal me ma_ta;
el que tan mui_to tar_da,
se ros_tro ou_tro lho ca_ta,
er_go o meu, que_rí_a
mor_rer hoj'es_te dí_a.

ca meu da_no se_rí_a
de vi_ver má_is un dí_a.
[fin]
