[FERNÁN_RODRÍGUEZ_DE_CALHEIROS][Trovadorismo]
[inicio]
ma_dre, pas_sou per aq_uí un ca_va_lei_ro
e lei_xou-me na_mo_rad'e con mar_tei_ro:
ai, ma_dre, os seus a_mo_res hei;
se me los hei,
ca mi os bus_quei,
ou_tros me lhe dei;
ai, ma_dre, os seus a_mo_res hei.

ma_dre, pas_sou per aq_uí un fil_lo d'al_go
e lei_xou-m'as_sí pe_na_da, com'eu an_do:
ai, ma_dre, os seus a_mo_res hei;
se me los hei,
ca mi os bus_quei,
ou_tros me lhe dei;
ai, ma_dre, os seus a_mo_res hei.

ma_dre, pas_sou per aq_uí quen non pas_sas_se
e lei_xou-m'as_sí pe_na_da, mais lei_xas_se:
ai, ma_dre, os seus a_mo_res hei;
se me los hei,
ca mi os bus_quei,
ou_tros me lhe dei;
ai, madr', os seus a_mo_res hei.
[fin]
