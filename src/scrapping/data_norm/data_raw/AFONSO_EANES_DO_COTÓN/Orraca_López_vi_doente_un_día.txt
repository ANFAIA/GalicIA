or_ra_ca ló_pez vi do_en_te un dí_a
e pre_gun_tei-a se gua_re_ce_rí_a.
e dis_se-m'e_la, tod'en jo_gra_rí_a:
«sõ_o vel_ha e cuid'a gua_re_cer».

e di_xe-lh'eu: «cui_da_des gran fo_lí_a,
ca i má_is vej'eu das vel_has mor_rer».

e di_xe-lh'eu: «gran fo_lí_a pen_sa_des,
se per vel_he_ce a gua_re_cer cui_da_des;
pe_ro non vos di_gu'eu que non vi_va_des
quan_to vos deus qui_ser lei_xar vi_ver;
mais en vel_hi_ce non vos a_tre_ve_des,
ca i má_is vej'eu das vel_has mor_rer».
[fin]
