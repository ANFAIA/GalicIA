[JOÁN_ZORRO][Trovadorismo]
[inicio]
quen vis_se an_dar 
fre_mo_sin_ha,
com'eu vi, d'a_mor coi_ta_da
e tan mui_to na_mo_ra_da
que, cho_ran_do as_sí di_zí_a:
«ai a_mor,
lei_xe_des-m'ho_je
de so-lo ra_mo 
fol_gar
e de_pois trei_des-vos 
mi_go
meu a_mi_go de_man_dar».

quen vis_se an_dar a fre_mo_sa,
com'eu vi, d'a_mor cho_ran_do
e di_zen_do e ro_gan_do,
por a_mor da glo_rio_sa:
«ai a_mor,
lei_xe_des-m'ho_je
de so-lo ra_mo 
fol_gar
e de_pois trei_des-vos 
mi_go
meu a_mi_go de_man_dar».

quen lhi viss'an_dar fa_zen_do
quei_xu_mes d'a_mor d'a_mi_go,
que a_ma e, sem_pre si_go
cho_ran_do, as_sí di_zen_do:
«ai a_mor,
lei_xe_des-m'ho_je
de so-lo ra_mo 
fol_gar
e de_pois trei_des-vos 
mi_go
meu a_mi_go de_man_dar».
