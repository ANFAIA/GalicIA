cac_tus e_ra un cen_cer_ro de de_ser_to
dor_mi_ta_ba en_tre tor_men_tas de a_re_a e frí_o
sin_ce_rá_ba_se co ven_to de ou_to_no e bá_go_as
a_li_men_tá_ba_se do ar con_scen_te da i_ner_cia
que su_pu_ña an_clar_se na ter_ra in_fér_til
e so_por_ta_ba co_mo po_dí_a os seus pin_chos
que da_ban a vol_ta e fa_cí_a_no san_grar
ás ve_ces de fe_li_ci_da_de, pe_ro sen can_ci_ón
que con_ver_ti_ra a a_re_a en ve_re_das ver_des
que lle o_tor_ga_ran de no_vo o tan ne_ga_do
a ca_pa_ci_da_de de re_en_con_trar_se co a_mor.
[fin]
