[i_ni_cio]
non i_mos ma_tar mou_ros,
meus ir_máns, por fa_vor,
(a_ín_da que o con_sel_lo nes_tes tem_pos
ca_re_ce de va_lor)
non i_mos, non, tras un ca_ba_lo bran_co
e un_ha es_pa_da ful_xen_te co_ma o sol,
un e ou_tra guia_dos
po_lo bra_zo for_zu_do dun si_ñor
que a_mo_ra_do dos i_be_ros guer_rei_ros
as de_li_cias do ceu a_ban_do_nou

dei_xe_mos que as me_se_tas,
ir_tas e trig_uei_ras,
er_gan ro_xas ban_dei_ras
a toq_ue de trom_pe_tas,
que pra nós a_bon_da
un sin_xe_lo al_tar
nun_ha fur_na fon_da

ou na so_be_ra_na

que o pa_so le_ne dun via_xei_ro
con san_gue quen_te dei_xou mar_ca_da.

e ho_xe fa_ce_mos fes_ta
no meio do ca_mi_ño,
sin de_sar_re_ar a bes_ta
nin be_ber moi_to vi_ño;

sin men_tar va_na_men_te a va_len_tí_a
