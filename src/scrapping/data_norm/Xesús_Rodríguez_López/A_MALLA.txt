pois n' a ei_ra que pe_ga_da
por un la_do á ca_sa es_tá

con ti_ras de cor_re_al,

un_ha su_be_la, un_ha vil_la,
cra_vos de to_dal-as cras,
un mar_te_lo, un_has ti_xei_ras
muy en_fer_ru_xa_das xa
e, pr' a_bran_dar mais os coi_ros,
un pou_co lar_do sin sal.

pois ti_ña moi_tos a_mi_gos
tod' a xen_te do tio brais,
por ser sem_pre moy do_í_da
pra cal_quer ne_ce_si_dad.

de ca_ra á e_la al_gús van.
