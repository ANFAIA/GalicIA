[i_ni_cio]
xa se sa_be que o pan que se mal_la
vay por es_tes pa_sos:
ten_den, mal_lan, a ai_ra_da re_vol_ven,
e vol_ta á mal_la_lo.

en des_pois_xas es_pal_lan y-en fei_xes
a pal_la qu' é lar_ga
van a_mon_to_nan_do,
qu' ó pal_lei_ro al_gus ho_mes a le_van,
na_men_tras qu' en tan_to
ou_tros xun_tan a pal_la mi_ú_da
con mal_les y-an_ga_zos
y-os co_a_ños a_xun_tan n' un ver_bo
con se_cos ra_mal_los.

co_mo o sol y o mal_lar tod' o dí_a
dan moi_to can_sa_ce_o,
en_tre ca_da mal_la_da des_can_san
á som_bra dei_ta_dos,
un pi_to fa_cen_do,
ou rin_do, ou ruan_do.
