[Xesús_Rodríguez_López][Poesía contemporánea]
[inicio]
ó xen_tar to_dos be_ben o vi_ño
por un mes_mo va_so,
y-a co_mi_da re_ma_tan con cun_cas
de lei_te ma_za_do.

Y-as ra_pa_zas que po_den ós mo_zos
col_ler des_cui_da_dos,
fan sal_tar_lles o lei_te d' as cun_cas
ti_rán_dol_les dren_to
de pan uns a_na_cos;
di_ver_si_ón que dá sem_pre mo_ti_vo
pra di_tos e con_tos
que mais val ca_la_los.
[fin]
