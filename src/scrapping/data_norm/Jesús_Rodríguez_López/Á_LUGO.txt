[Jesús_Rodríguez_López][Poesía contemporánea]
[inicio]
a tí, que_ri_do pue_blo mí_o, que re_cuer_das mi
ni_ñez, con sus in_com_pa_ra_bles a_le_grí_as y mi ju_ven_tud en con_stan_te luc_ha
con ias as_pi_ra_cio_nes del co_ra_zón; á tí, que guar_das ias ce_ni_zas de
mis vir_tuo_sos pa_dres, ias que son pa_ra mí un te_so_ro de ve_ne_ra_ci_ón y
res_pe_to; á tí, don_de tan_to he a_ma_do y en don_de tan_to he su_fri_do; á
tí, á quien tan_to de_bo y a_ún más quie_ro, á pe_sar de los pe_sa_res, á
tí de_di_co es_te re_cuer_do por_que en tí es_tán com_pen_dia_dos los más
ca_ros a_fec_tos de mi vi_da.
[fin]
