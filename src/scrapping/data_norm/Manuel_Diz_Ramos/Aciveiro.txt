[Manuel_Diz_Ramos][Poesía contemporánea]
[inicio]
en_tre va_les e bar_ran_cos
cur_vas e gar_gan_tas
con_stru_í_ron mo_nas_te_rios
pa_ra ve_ne_rar ós san_tos.

mos_tei_ros, er_mi_tas e pa_zos
en_tre o ter_re_o e re_li_xio_so
pa_ra en_si_nar a ca_teq_ui_zar
do hu_mil_de ó po_de_ro_so.

o mos_tei_ro de a_ci_vei_ro
a sú_a mi_si_ón e_ra un a_fán
es_cri_bir a ver_da_dei_ra his_to_ria
a_gac_ha_da no mon_te can_dán.

nes_tes mon_tes de cli_ma re_cio,
nen san_tos, sa_bios ou ma_xis_ter,
nen gran_des guer_rei_ros co_ñe_ci_dos
pe_ro, la_bra_do_res e mi_sio_na_rios do cís_ter.

ver_des pra_dos con flo_ri_dos ma_tor_ra_les,
den_sas fra_gas de car_bal_los e cas_ti_ñei_ros.
mon_tes con ou_tei_ros e ru_ti_lan_tes pe_ñas_ca_les.
car_ras_quei_ras e to_xos no me_dio dos pi_ñei_ros.
[fin]
