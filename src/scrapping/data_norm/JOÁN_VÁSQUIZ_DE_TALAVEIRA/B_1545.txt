di_rei-vos o_ra que o_í di_zer
de ma_rí_a le_ve, as_sí ha_ja ben,
po_la man_ce_ba, que se de_sa_vén
de_la; e, pois lh'a_lí non quer vi_ver,
e_na mo_e_da vel_ha vai mo_rar
do_na ma_rí_a le_ve, a seu pe_sar;

ca a_tal do_na com'e_la gua_rir
non pod'a_lí, se man_ce_ba non ha;
e ve_des que o_í, a_mi_gos, ja:
que, pois que se lh'a man_ce_ba quer ir,
e_na mo_e_da vel_ha vai mo_rar
do_na ma_rí_a le_ve, a seu pe_sar;

ca diz que mo_ra_ra a_lí mal, ai!,
poi-la man_ce_ba si_go non hou_ver,
e con_tra san mar_tin_ho mo_rar quer;
po_la man_ce_ba que xi lh'o_ra vai,
e_na mo_e_da vel_ha vai mo_rar
do_na ma_rí_a le_ve, a seu pe_sar;

ca non pod'a man_ce_ba es_cu_sar,
se na mo_e_da vel_ha non mo_rar.
