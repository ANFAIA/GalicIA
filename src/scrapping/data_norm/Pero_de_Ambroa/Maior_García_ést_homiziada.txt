[i_ni_cio]
maior gar_cí_a ést'ho_mi_zia_da,
que tan_to guar_da seu cor_po, de pran,
que ja de noi_te nun_ca e_la man,
co_mo as ou_tras, na sú_a pou_sa_da;
e guar_da-se, ja nun_ca tan_to vi:
ca, u man ho_je, non mar_rá cras i,
des i, de noi_te, nun_ca dor_me na_da.

e, con to_do es_to, as_sí é cus_tuia_da,
que non po_de ho_me sa_ber seu ta_lán,
e en má_is de cen lo_ga_res a ve_rán
dei_tar, mais pou_co é i sa fi_ca_da;
e, u al_ber_ga, di_rei-vos o que faz:
nun_ca ho_men në_ún na ca_sa jaz
que o a me_os non ca_te ü_a vez.

e an_da-s'e_la tan des_se_gu_ra_da,
co_me se nun_ca i pren_des_se a_fán;
non ést'as_sí, ca mui_tos vos di_rán
que ben mil ve_ces lle me_ten cia_da,
e non a po_de nulh'ho_men col_her
na mã_o, se i a non ac_ha ja_zer,
ou quan_do'sté, ou quan_do é le_van_ta_da.

e non a po_de_rá ho_men ac_har
se_nón cuan_do se qui_ser le_van_tar,
ou a_o se_rán ou a_a ma_dru_ga_da?
[fin]
